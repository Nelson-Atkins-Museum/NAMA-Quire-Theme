{{/* A simple partial to create a list of contributor names. Looks first for .full_name and otherwise combines first and last. Expects three values to be passed in: range The range of contributors for the list. For example, .Site.Data.publication.contributor or .Params.contributor contributorType Can be "primary", "secondary", "project-team", or "all" listType Can be "string", "list", "list-plus". "string" creates a list of names with proper "and" or "," connectors. "list" creates and unordered list of names, and "list-plus" adds titles and affiliations when available. Site Usually just $.Site. Used to pass in context in order to grab contributor info from .Site.Data.publication by ID. EXAMPLE PARTIAL USE: partial "contributor-list.html" (dict "range" .Params.contributor "contributorType" "primary" "listType" "string" "Site" $.Site) */}} {{- if ne .range nil }} {{- if eq .listType "string" }} {{- if eq .contributorType "all" }} {{- $len := len .range -}} {{- range $index, $element := .range -}} {{- if eq $len 1 -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else if eq (add $index 1) $len -}} and {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else -}} {{- if eq $len 2 -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }}, {{- end -}} {{- end -}} {{- end }} {{- else }} {{- $len := len (where .range "type" .contributorType) -}} {{- range $index, $element := where .range "type" .contributorType -}} {{- if eq $len 1 -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else if eq (add $index 1) $len -}} and {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else -}} {{- if eq $len 2 -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- else -}} {{ template "name-check" (dict "Site" $.Site "contributor" .) }}, {{- end -}} {{- end -}} {{- end }} {{- end }} {{- else if eq .listType "list" }} {{- if eq .contributorType "all" }}
    {{- range .range -}}
  • {{ template "name-check" (dict "Site" $.Site "contributor" .) }}
    • {{- end }}
{{- else }}
    {{- range where .range "type" .contributorType -}}
  • {{ template "name-check" (dict "Site" $.Site "contributor" .) }}
    • {{- end }}
{{- end }} {{- else if eq .listType "list-plus" }} {{- if eq .contributorType "all" }}
    {{- range .range -}}
  • {{- template "name-check" (dict "Site" $.Site "contributor" .) -}} {{- if or .title .affiliation -}} {{- template "title" . -}} {{- else if .id }} {{- range where $.Site.Data.publication.contributor "id" .id -}} {{- template "title" . -}} {{- end -}} {{- end -}}
    • {{- end }}
{{- else }}
    {{- range where .range "type" .contributorType -}}
  • {{ template "name-check" (dict "Site" $.Site "contributor" .) }} {{- if or .title .affiliation -}} {{- template "title" . -}} {{- else if .id }} {{- range where $.Site.Data.publication.contributor "id" .id -}} {{- template "title" . -}} {{- end -}} {{- end -}}
    • {{- end }}
{{- end }} {{- end }} {{- end }} {{- define "name-check" -}} {{- if or .contributor.full_name (and .contributor.first_name .contributor.last_name) -}} {{- template "name" .contributor -}} {{- else if .contributor.id -}} {{- range where $.Site.Data.publication.contributor "id" .contributor.id }}{{ template "name" . }}{{ end -}} {{- end -}} {{- end -}} {{- define "name" -}} {{- if .full_name -}} {{ .full_name }} {{- else -}} {{ .first_name }} {{ .last_name }} {{- end -}} {{- end -}} {{- define "title" -}} {{- if or .title .affiliation}}, {{ end -}} {{- with .title }}{{ . }}{{ end -}} {{- if and .title .affiliation }}, {{ end -}} {{- with .affiliation }}{{ . }}{{ end -}} {{- end -}}
